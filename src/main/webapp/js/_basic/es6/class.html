Before ES6, JavaScript only used prototype pattern or function to simulate classical pattern.

the old way:

        function MyClass() {
            this.a = 'first';
        };
        
        MyClass.prototype.sayHi = function() {
            console.log(this.a) ;  //outpot 'first'
        };

Now Es6 support Class and Inheritance(more and more close to Java):

        class Person {
            name = "";
            constructor(name) {
                this.name = name;            
            }
            whoAreYou() {
                return `Hi i'm ${this.name}`;
            }
        }
        let person = new Person('Justin');
        console.log(person.whoAreYou());

        class Student extends Person { 
            course;
            constructor(name, course) {
                super(name);
                this.course = course;
            }
            whoAreYou() { 
                return `${super.whoAreYou()} and i'm studying ${this.course}`;
            }
        }

        let student = new Student('Justin', 'Computer Science');
        console.log(student.whoAreYou());

Please open console to see the result

All browsers except IE support it
